	@staticmethod
	def importance_to_color(importance: int) -> str:
		"""
			Maps an importance value to a color code.
			
			:param importance: The importance value to be converted to a color code.
			:return: A string representing the color code.
			
			:example:

				>>> Cap.importance_to_color(5)
				'"#00FF0050"'
		
		"""
		
		# Return red if importance is 0
		if importance <= 0.1:
			return '"#FF0000"'
		
		normalized_importance = int(importance / 10 * 255)  # Weight backlinks

		# Convert to hexadecimal and pad with zeros
		hex_importance = hex(normalized_importance)[2:].zfill(2)
		
		return f'"#00FF00{hex_importance}"' # Return color code
    


	@performance.timeit
	def plot_tree_diagram(self, root_doc: str, max_depth: int=1, input_data_path="tests/_outputs/", output_data_path="tests/_outputs/") -> Image:
		"""
		Plot a tree diagram based on the data from the provided root document.

		:param root_doc: The root document from which to start the tree.
		:param max_depth: The maximum depth of the tree (default is 1).
		:param input_data_path: The path to the input data files (default is "tests/_outputs/").
		:param output_data_path: The path to the output data files (default is "tests/_outputs/").

		:return: A tree diagram visualization.

		:example:

			>>> cap = Cap()
			>>> cap.plot_tree_diagram('root_document')
   
		"""

		# Load nodes and edges data
		try:
			nodes_df = pd.read_csv(f'{input_data_path}nodes.csv') 
			edges_df = pd.read_csv(f'{input_data_path}edges.csv')  
		except FileNotFoundError as e:
			print(f"File not found: {e}")
			return

		# Create root node and add children
		root = self.MyNode(root_doc)
		root.add_children(nodes_df, edges_df, max_depth=max_depth)

		# Create Dot file
		DotExporter(root, nodeattrfunc=lambda node: f'color={self.importance_to_color(node.importance)}').to_dotfile(f"{output_data_path}tree.dot")

		# Generate png image from dot file
		os.system(f'dot -Tpng {output_data_path}tree.dot -o {output_data_path}tree.png')  
		os.remove(f"{output_data_path}tree.dot")  # Remove dot file
  
		# Display image in Jupyter notebook
		return Image(filename=f'{output_data_path}tree.png', width=1000, height=1000, retina=True, unconfined=False)

	class MyNode(Node):
		""" 
  			A custom Node class that extends the base Node class and adds an importance attribute. 
     
     	"""

		def __init__(self, name, parent=None, children=None, importance=1):
			"""
			Initialize the Node with a name, optional parent and children, and an importance value.

			Args:
				name (str): The name of the node.
				parent (Node, optional): The parent of the node. Defaults to None.
				children (list of Node, optional): The children of the node. Defaults to None.
				importance (int, optional): The importance value of the node. Defaults to 1.

			The Node class represents a node in a tree structure. Each node has a name, an optional parent node, zero or more child nodes, and an importance value. The importance value is used to determine the relative importance of the node in the tree.
   
			"""
   
			super().__init__(name, parent, children)
			self.importance = importance

		def add_children(self, nodes_df: pd.DataFrame, edges_df: pd.DataFrame, depth: int=1, max_depth: int=2):
			"""
			Add children to the node from the nodes and edges dataframes, up to a specified maximum depth.

			Args:
				nodes_df (DataFrame): A dataframe containing the nodes data.
				edges_df (DataFrame): A dataframe containing the edges data.
				depth (int): The current depth of the node.
				max_depth (int): The maximum depth to which children should be added.

			This method adds child nodes to the current node based on the nodes and edges dataframes. The method recursively adds children up to the specified maximum depth. If the maximum depth is reached, no more children are added.
   
			"""			# If the current depth is greater than the maximum depth, stop recursion
   
			if depth > max_depth:
				return

			# Get the URLs of the child nodes
			child_urls = self.get_child_ids(self.name, nodes_df, edges_df)

			# If the maximum depth is greater than 1 and the current depth is 2 or more,
			# limit to the first child URL
			if max_depth > 1 and depth >= 3:
				child_urls = child_urls[:2]  # Only take the first two children

			# For each child URL, create a new MyNode instance and add it as a child of the current node
			for child_url in child_urls:
				child = nodes_df.loc[nodes_df['url'] == child_url].iloc[0]
				child_node = Cap.MyNode(child_url, parent=self, importance=child.importance)
				child_node.add_children(nodes_df, edges_df, depth=depth+1, max_depth=max_depth)

		def get_child_ids(self, url, nodes_df: pd.DataFrame, edges_df: pd.DataFrame):
			"""
			Get the child IDs for a given URL from the nodes and edges dataframes.

			Args:
				url (str): The URL of the node.
				nodes_df (pd.DataFrame): A dataframe containing the nodes data.
				edges_df (pd.DataFrame): A dataframe containing the edges data.

			Returns:
				pd.Series: A series containing the URLs of the child nodes.

			This method finds the node in the nodes dataframe that matches the given URL, then finds the edges in the edges dataframe that have the matching node as the source. The URLs of the target nodes of these edges are returned as a series.
			
   			"""
      
			matching_nodes = nodes_df[nodes_df['url'] == url].index
			if matching_nodes.empty:
				return pd.Series(dtype='str')
			node_id = matching_nodes[0]
			child_ids = edges_df.loc[edges_df['source'] == node_id, 'target']
			child_urls = nodes_df.loc[child_ids, 'url']
			return child_urls
