    def find_neighbors(self, start: str, method: str = 'BFS', curr_node: str = None):
        """Performs a search of the graph using either breadth-first search (BFS) or depth-first search (DFS).

            :param start: The node to be searched for.
            :param method: The search method to use ('BFS' or 'DFS'). Default is 'BFS'.
            :param curr_node: The current node.

            :return: A list of nodes that are connected to the input node.

            :example:

                >>> search(start, method, curr_node)

        """

        visited = set()  # Initialize a set to keep track of visited nodes

        if method == 'BFS':
            queue = [start]  # Initialize a queue with the starting node

            # Continue until the queue is empty
            while queue:
                node = queue.pop(0)  # Remove the first node from the queue

                if node not in visited:
                    visited.add(node)  # Mark the node as visited
                    # Add the unvisited neighbors of the node to the queue
                    queue.extend(self.graph[node] - visited)

                # Print the node
                print(node, end=' ')

        elif method == 'DFS':
            def dfs(node):
                visited.add(node)  # Mark the current node as visited
                print(node, end=' ')  # Print the current node

                for neighbour in self.graph[node]:
                    # If the neighbor has not been visited yet, recursively perform a depth-first search on the neighbor
                    if neighbour not in visited:
                        dfs(neighbour)

            dfs(start)

        # Mark the current node as finished processing
        if curr_node:
            curr_node.finished_processing = True

    @performance.timeit
    def partition_data(self):
        """
            This function partitions the graph into subgraphs, each containing a subset of the nodes. 
            The partitioning is done based on the modulus of the node's ID and the total number of nodes. 
            This ensures a balanced distribution of nodes across the partitions.

            :return: A dictionary where the keys are the partition numbers and the values are lists of nodes belonging to that partition.

            :example:

                >>> partitioned_graph = partition_data()

        """
        
        # Initialize an empty dictionary to hold the partitioned graph
        partitioned_graph = {}
        
        # Iterate over each node in the graph
        for node in self.nodes:
            # Calculate the partition number for the node
            partition = node.id % self.num_nodes
            
            # If the partition number is not already a key in the dictionary, add it with an empty list as its value
            if partition not in partitioned_graph:
                partitioned_graph[partition] = []
            
            # Add the node to the list of nodes in its partition
            partitioned_graph[partition].append(node)
        
        # Return the dictionary representing the partitioned graph
        return partitioned_graph

@performance.timeit
def query_tree():
    """
    Endpoint for handling Mycelium database requests.
    
    This function receives a JSON payload from a POST request and passes it to the handler function.
    The result returned by the handler function is then converted to JSON and returned as the response.
    
    """
    try:
        # Retrieve the JSON payload from the request
        data = request.get_json()
        path = request.path

        # Create an instance of GraphDB
        db = GraphDB(data.get("meta", {}))

        # See the functions for the details
        if path == '/v1/db-query':
            func = db.query
        elif path == '/v1/db-insert':
            func = db.insert
        elif path == '/v1/db-update-nightly':
            func = db.update_nightly
        elif path == '/v1/db-machine-state':
            func = db.get_state
        else:
            raise ValueError(f"Missing required parameter: {path}")

        return jsonify(func(data))
    except Exception as e:
        print(f"An unexpected error occurred: {e}")
        return jsonify({
            "status": 500,
            "error": f"An error occurred: {traceback.format_exc()}",
        })
