class PosEmbeds:
    def __init__(self, config):
        """
            The Embeddings class wraps a list of embeddings used to encode different types of positional information for HTML tags. 

            It includes the following embeddings:
                1. node_embeddings: Embeds the index of the current node.
                2. parent_embeddings: Embeds the index of the parent of the current node.
                3. sibling_embeddings: Embeds the index of the current node with respect to its siblings.
                4. depth_embeddings: Embeds the depth of the node.
                5. tag_embeddings: Embeds the <tag> of the node.

            :param config: A dictionary of configuration parameters.

            :return: A tuple containing all the embeddings.
            
        """
        self.node_embeddings = tf.Variable(tf.random.uniform([config.node_size, config.embed_size]))
        self.parent_embeddings = tf.Variable(tf.random.uniform([config.parent_size, config.embed_size]))
        self.sibling_embeddings = tf.Variable(tf.random.uniform([config.sibling_size, config.embed_size]))
        self.depth_embeddings = tf.Variable(tf.random.uniform([config.depth_size, config.embed_size]))
        self.tag_embeddings = tf.Variable(tf.random.uniform([config.tag_size, config.embed_size]))

    @performance.timeit
    def forward(self, node_ids=None, parent_node_ids=None, sibling_node_ids=None, depth_ids=None, tag_ids=None):
        """
          A method that returns a tuple containing all the embeddings.

          :param node_ids: The index of the current node.
          :param parent_node_ids: The index of the parent of the current node.
          :param sibling_node_ids: The index of the current node with respect to its siblings.
          :param depth_ids: The depth of the node.
          :param tag_ids: The <tag> of the node.
          
          :return: A tuple containing all the embeddings.
          
        """
        
        embeddings = (        
            tf.nn.embedding_lookup(self.node_embeddings, node_ids) +
            tf.nn.embedding_lookup(self.parent_embeddings, parent_node_ids) +
            tf.nn.embedding_lookup(self.sibling_embeddings, sibling_node_ids) +
            tf.nn.embedding_lookup(self.depth_embeddings, depth_ids) +
            tf.nn.embedding_lookup(self.tag_embeddings, tag_ids)
        )
        return embeddings

    def __init__(self):
        """
        This code first defines a forward function that computes the output of the self-attention mechanism in the Transformer. 
        """
            # Assume we have some pre-initialized weights
        self.W_q = tf.Variable(tf.random.normal((5, 5)))
        self.W_k = tf.Variable(tf.random.normal((5, 5)))
        self.W_v = tf.Variable(tf.random.normal((5, 5)))

    @performance.timeit
    def positional_encoding(self, position: int, d_model: int):
        """
        This method generates a matrix of angles for positional encoding.

        :param position: The position of the node in the tree.
        :param d_model: The dimension of the model.

        :return: The positional encoding.

        :example:

            >>> from mycelium.model import Transformer
            >>> Transformer.positional_encoding(1, 2)

        """

        # Generate a matrix of angles for positional encoding
        angle_rads = self.get_angles(tf.range(position)[:, tf.newaxis],
                                tf.range(d_model)[tf.newaxis, :],
                                d_model)
        
        # Apply sin function to even indices in the array (2i)
        angle_rads[:, 0::2] = tf.math.sin(angle_rads[:, 0::2])
        
        # Apply cos function to odd indices in the array (2i+1)
        angle_rads[:, 1::2] = tf.math.cos(angle_rads[:, 1::2])
        
        # Add an extra dimension to match the batch size
        pos_encoding = angle_rads[tf.newaxis, ...]
        
        # Return the positional encoding
        return pos_encoding

    @performance.timeit
    def get_angles(self, pos: int, i: int, d_model: int):
        """This method computes the angle rates for positional encoding.

            :param pos: The position of the node in the tree.
            :param i: The dimension of the model.
            :param d_model: The dimension of the model.

            :return: The angle rates for positional encoding.

            :example:

                >>> from mycelium.model import Transformer
                >>> Transformer.get_angles(1, 2, 3)

        """

        # Calculate the angle rates for positional encoding
        angle_rates = 1 / tf.pow(10000, (2 * (i//2)) / tf.cast(d_model, tf.float32))
        return pos * angle_rates

    @performance.timeit
    def forward(self, X: tf.Tensor, W_q: tf.Tensor, W_k: tf.Tensor, W_v: tf.Tensor):
        """This method computes the output of the self-attention mechanism in the Transformer.

            :param X: The input.
            :param W_q: The query matrix.
            :param W_k: The key matrix.
            :param W_v: The value matrix.

            :return: The output of the self-attention mechanism.

            :example:
            
                >>> from mycelium.model import Transformer
                >>> Transformer.forward(X, W_q, W_k, W_v)

        """

        # Compute Q, K, V
        Q = tf.matmul(X, W_q)
        K = tf.matmul(X, W_k)
        V = tf.matmul(X, W_v)

        # Compute attention scores
        scores = tf.matmul(Q, K, transpose_b=True) / tf.sqrt(float(K.shape[-1]))

        # Apply softmax to get attention weights
        weights = tf.nn.softmax(scores)

        # Compute output
        out = tf.matmul(weights, V)
        
        return out, weights
