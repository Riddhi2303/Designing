def gen_hex_from_cm(name):
    cmap = cm.get_cmap(name)
    return [colors.rgb2hex(cmap(i)) for i in range(cmap.N)]

        @staticmethod
        def vertical_dot_plot(categories: List, data: List[List[float]], **optional_params) -> go.Figure:
            """
                Creates a vertical dot and whisker plot.

                :param categories: Categories to place along the x-axis.
                :param data: The data for each category.
                :param optional_params:
                    * colors: A list of colors for each category. Defaults to a DefaultColors.primary_line if not specified.
                    * xlabel: Label for the x-axis. Defaults to an empty string if not specified.
                    * ylabel: Label for the y-axis. Defaults to an empty string if not specified.
                    * title: Title of the plot. Defaults to empty strin if not specified.
                    * theme: Theme for the plot. Defaults to 'plotly' if not specified.
                :return: A plotly figure object
                
                :example:
                
                    >>> from plotting import *
                    >>> dash_manager = DashManager()
                    >>> plot_generator = PlotGenerator.Rank
                    >>> categories = ['Category A', 'Category B', 'Category C']
                    >>> data = [\
                        [10, 12, 14, 13, 12, 13, 12, 10, 14, 11],  # Data for Category A\
                        [15, 14, 16, 15, 15, 15, 15, 25, 17],  # Data for Category B\
                        [20, 22, 11, 1, 20, 21, 20, 9, 21, 23]  # Data for Category C\
                    ]
                    >>> fig = plot_generator.vertical_dot_plot(categories, data, title='My Dot and Whisker Plot',\
                                      xlabel='My Categories',\
                                      ylabel='Value')
                    >>> dash_manager.add_figure(_fig)
                    >>> dash_manager.run()

                    .. image::   ../../../assets/plot_vertical_dot.png
            
            """
            
            assert len(categories) == len(
                data), "Length of data and categories should be equal; found {len(data)} and {len(categories)}"
            # Initialize a figure
            _fig = go.Figure()
            # Determine colors for each category if provided, otherwise default to a single color
            colors = optional_params.get('colors', [DefaultColors.primary_line] * len(categories))

            # Add whiskers and median dots
            for i, (category, vals) in enumerate(zip(categories, data)):
                # Calculate the median and the whisker values (min, max)
                median_value = np.median(vals)
                whisker_min = st.t.interval(0.95, len(vals) - 1, loc=np.mean(vals), scale=st.sem(vals))[0]
                whisker_max = st.t.interval(0.95, len(vals) - 1, loc=np.mean(vals), scale=st.sem(vals))[1]

                # Add median dots
                _fig.add_trace(go.Scatter(
                    x=[i] * len(vals),
                    y=vals,
                    mode='markers',
                    marker=dict(
                        color=DefaultColors.background_dots,
                        size=optional_params.get('dot_size', 5),
                        opacity=0.5
                    ),
                    name=category,
                    showlegend=False
                ))

                _fig.add_trace(go.Scatter(
                    x=[i],
                    y=[np.median(vals)],
                    mode='markers',
                    marker=dict(
                        color=colors[i % len(colors)],
                        size=optional_params.get('dot_size', 9),
                        opacity=0.5
                    ),
                    name="median",
                    showlegend=False
                ))

                # Add the whiskers as lines
                _fig.add_trace(go.Scatter(
                    x=[i, i],
                    y=[whisker_min, whisker_max],
                    mode='lines',
                    line=dict(color=colors[i % len(colors)], width=2, ),
                    showlegend=False
                ))

                # Add whisker caps
                cap_width = optional_params.get('cap_width', 0.01)
                _fig.add_trace(go.Scatter(
                    x=[i - cap_width, i + cap_width, None, i - cap_width, i + cap_width],
                    y=[whisker_min, whisker_min, None, whisker_max, whisker_max],
                    mode='lines',
                    line=dict(color=DefaultColors.primary_line, width=2, ),
                    showlegend=False
                ))

            # Update layout
            _fig.update_layout(
                xaxis=dict(
                    title=optional_params.get('xlabel', ''),
                    tickmode='array',
                    tickvals=list(range(len(categories))),
                    ticktext=categories
                ),
                yaxis=dict(title=optional_params.get('ylabel', '')),
                title=optional_params.get('title', ''),
                template=optional_params.get('theme', 'plotly')
            )

            return _fig

        def horizontal_dot_plot(categories: List, data: List[List], **optional_params):
            """
                Creates a horizontal dot and whisker plot.

                :param categories: Categories to place along the x-axis
                :param data: The data for each category.
                :param optional_params:
                    * colors: A list of colors for each category. Defaults to a primary color line if not specified.
                    * xlabel: Label for the x-axis. Defaults to an empty string if not specified.
                    * ylabel: Label for the y-axis. Defaults to an empty string if not specified.
                    * title: Title of the plot. Defaults to empty strin if not specified.
                    * theme: Theme for the plot. Defaults to 'plotly' if not specified.
                :return: A plotly figure object

                :example:
                
                    >>> dash_manager = DashManager()
                    >>> plot_generator = PlotGenerator.Rank()
                    >>> categories = ['Category A', 'Category B', 'Category C']
                    >>> data = [\
                        [10, 12, 14, 13, 12, 13, 12, 10, 14, 11],  # Data for Category A\
                        [15, 14, 16, 15, 15, 15, 15, 25, 17],  # Data for Category B\
                        [20, 22, 11, 1, 20, 21, 20, 9, 21, 23]  # Data for Category C\
                    ]
                    >>> fig = plot_generator.horizontal_dot_plot(categories, data, title='My Dot and Whisker Plot',\
                                    xlabel='My Categories',\
                                    ylabel='Value')
                    >>> dash_manager.add_figure(_fig)
                    >>> dash_manager.run()

                    .. image::   ../../../assets/plot_horizontal_dot.png
                    
            """
            
            assert len(categories) == len(data), (f"Length of data and categories should be equal; found {len(data)} "
                                                  f"and {len(categories)}")

            # Initialize a figure
            _fig = go.Figure()

            # Determine colors for each category if provided, otherwise default to a single color
            colors = optional_params.get('colors', [DefaultColors.primary_line] * len(categories))

            # Add whiskers and median dots for each category
            for i, (category, vals) in enumerate(zip(categories, data)):
                # Calculate the median and the confidence interval
                median_value = np.median(vals)
                conf_int = st.t.interval(0.95, len(vals) - 1, loc=np.mean(vals), scale=st.sem(vals))

                # Color for the current category
                color = colors[i % len(colors)]

                _fig.add_trace(go.Scatter(
                    y=[i] * len(vals),
                    x=vals,
                    mode='markers',
                    marker=dict(
                        color=DefaultColors.background_dots,
                        size=optional_params.get('dot_size', 5),
                        opacity=0.5
                    ),
                    name=category,
                    showlegend=False
                ))

                # Add whiskers as lines
                _fig.add_trace(go.Scatter(
                    y=[i, i],
                    x=conf_int,
                    mode='lines',
                    line=dict(color=color, width=4),
                    showlegend=False
                ))

                # Add whisker caps
                cap_size = optional_params.get('cap_size', 0.05)
                _fig.add_trace(go.Scatter(
                    y=[i - cap_size, i + cap_size, None, i - cap_size, i + cap_size],
                    x=[conf_int[0], conf_int[0], None, conf_int[1], conf_int[1]],
                    mode='lines',
                    line=dict(color=color, width=4),
                    showlegend=False
                ))

                # Add median dots
                _fig.add_trace(go.Scatter(
                    y=[i],
                    x=[median_value],
                    mode='markers',
                    marker=dict(color=color, size=12),
                    showlegend=False
                ))

            # Update layout
            _fig.update_layout(
                yaxis=dict(
                    title=optional_params.get('ylabel', ''),
                    tickmode='array',
                    tickvals=list(range(len(categories))),
                    ticktext=categories,
                    automargin=True,
                ),
                xaxis=dict(
                    title=optional_params.get('xlabel', '')
                ),
                title=optional_params.get('title', ''),
                template=optional_params.get('theme', 'plotly'),
                showlegend=False
            )

            return _fig

        @staticmethod
        def horizontal_waterfall(categories: List, data: List, **optional_params) -> go.Figure:
            """
                Creates a horizontal waterfall chart.

                :param categories: Categories to be place along the y axis
                :param data: Data for each category
                :param optional_params:
                    * baseline: A baseline about which to flip the columns. Defaults to 0
                    * xlabel: Label for the x-axis. Defaults to an empty string if not specified.
                    * ylabel: Label for the y-axis. Defaults to an empty string if not specified.
                    * title: Title of the plot. Defaults to empty strin if not specified.
                    * theme: Theme for the plot. Defaults to 'plotly' if not specified.
                :return: A plotly figure object

                :example:
                
                    >>> dash_manager = DashManager()
                    >>> plot_generator = PlotGenerator.Rank()
                    >>> categories = ['Burger', 'Pizza', 'Sandwich', 'Pasta', 'Momos', 'Garlic']
                    >>> data = [-10, -5, -4, 4, 5, 10]
                    >>> fig = plot_generator.horizontal_waterfall(categories, values, \
                                            xlabel='Value', ylabel='Categories',\
                                            title='Horizontal Waterfall')
                    >>> dash_manager.add_figure(fig)
                    >>> dash_manager.run()

                .. image::   ../../../assets/plot_horizontal_waterfall.png
                
            """
            
            assert len(categories) == len(data), (f"Length of data and categories should be equal; found {len(data)} "
                                                  f"and {len(categories)}")

            # Set baseline value
            _baseline = optional_params.get('baseline', 0)

            # Combine categories and data
            pairs = list(zip(categories, data))
            if optional_params.get('sorted', True):
                pairs = sorted(pairs, key=lambda x: x[1])

            # Create a Plotly figure
            fig = go.Figure()

            # Plot the bars
            for category, value in pairs:
                # Determine the starting point and length of each bar
                if value > _baseline:
                    base = _baseline
                    length = value - _baseline
                else:
                    base = value
                    length = _baseline - value

                fig.add_trace(go.Bar(
                    y=[category],
                    x=[length],
                    base=base,
                    orientation='h',
                    marker=dict(color=DefaultColors.secondary_line if value < _baseline else DefaultColors.primary_line)
                ))

            # Set the layout for the figure
            fig.update_layout(
                xaxis_title=optional_params.get('xlabel', ''),
                yaxis_title=optional_params.get('ylabel', ''),
                title=optional_params.get('title', ''),
                showlegend=False
            )

            return fig

        def baseline_lineplot(baseline_series: List, data: List, categories: List, **optional_params: object) -> object:
            """
                Creates a baseine line plot.

                :param baseline_series: The base line series
                :param data: The other series to compare to the baseline
                :param categories: The categories for each series
                :param optional_params:
                    * colors: A list of colors for each category. Defaults to Pastel1 colormap if not specified.
                    * xlabel: Label for the x-axis. Defaults to an empty string if not specified.
                    * ylabel: Label for the y-axis. Defaults to an empty string if not specified.
                    * title: Title of the plot. Defaults to empty strin if not specified.
                    * theme: Theme for the plot. Defaults to 'plotly' if not specified.
                    * legend: Title of legend. Defaults to empty string.
                :return: A plotly figure object

                :example:
                
                    >>> dash_manager = DashManager()
                    >>> plot_generator = PlotGenerator.Deviation()
                    >>> baseline = [10, 20, 30]
                    >>> months = ['Jan', 'Feb', 'Mar']
                    >>> series_colors = ['red', 'green', 'blue']
                    >>> series1 = [12, 22, 32]
                    >>> series2 = [8, 18, 28]
                    >>> series3 = [15, 25, 35]
                    >>> data = [series1, series2, series3]
                    >>> fig = plot_generator.baseline_lineplot(baseline, data, months,  title='Sales Data', xlabel='Month', ylabel='Sales ($)', legend='Product')
                    >>> dash_manager.add_figure(_fig)
                    >>> dash_manager.run()

                .. image::   ../../../assets/plot_baseline.png
                
            """
            
            assert len(categories) == len(data), \
                (f"Length of data and categories should be equal; found {len(data)} and {len(categories)}")

            colors = optional_params.get('colors', gen_hex_from_cm('Pastel1'))

            # Create a Plotly figure
            _fig = go.Figure()

            # Add each series as a separate line trace
            for i, series in enumerate(data):
                percentage_change_series = [(value - baseline_series[j]) / baseline_series[j] * 100 for j, value in
                                            enumerate(series)]
                _fig.add_trace(go.Scatter(
                    x=categories,
                    y=percentage_change_series,
                    mode='lines+markers',
                    name=f'Product {i + 1}',
                    marker=dict(color=colors[i % len(colors)])
                ))

            # Add a horizontal line at y=0 for the baseline
            _fig.add_hline(y=0, line=dict(color=DefaultColors.baseline, dash='dash'))

            # Update layout based on optional parameters
            _fig.update_layout(
                xaxis_title=optional_params.get('xlabel', ''),
                yaxis_title=optional_params.get('ylabel', ''),
                title=optional_params.get('title', ''),
                legend_title=optional_params.get('legend', ''),
                template=optional_params.get('theme', 'plotly')
            )

            return _fig

        @staticmethod
        def scatter_plot(marker_shapes: List, x_axis_data: List[List], y_axis_data: List[List],
                         **optional_params) -> go.Figure:
            """
                Creates a scatter plot.

                :param marker_shapes: Marker shape of each series
                :param x_axis_data: The x axis value of each point of each series
                :param y_axis_data: The y axis value of each point of each series
                :param optional_params:
                    * xlabel: Label for the x-axis. Defaults to an empty string if not specified.
                    * ylabel: Label for the y-axis. Defaults to an empty string if not specified.
                    * title: Title of the plot. Defaults to empty strin if not specified.
                    * theme: Theme for the plot. Defaults to 'plotly' if not specified.
                :return: A plotly Figure object

                :example:
                
                    >>> dash_manager = DashManager()
                    >>> marker_shapes = ['square', 'circle', 'diamond']  # Plotly symbol names
                    >>> x_values = [list(range(10)), list(range(10)), list(range(10))]  # Example x-axis values for each group
                    >>> series_data = [\
                            [i ** 1.3 + np.random.rand() * 30.0 for i in range(10)],\
                            [i ** 1.5 + np.random.rand() * 40.0 for i in range(10)],\
                            [i ** 1.2 + np.random.rand() * 20.0 for i in range(10)]\
                        ]
                    >>> colors = ["#FC2964", "#0095FF", "#5C33F6"]
                    >>> correlation = PlotGenerator.Correlation()
                    >>> data = np.random.rand(len(y_axis_data), len(x_axis_data))
                    >>> fig = correlation.scatter_plot(marker_shapes, x_values, series_data, colors, \
                    title='Scatter Plot', xlabel='X-Axis', ylabel='Y-Axis')
                    >>> dash_manager.add_figure(fig)
                    >>> dash_manager.run()

                .. image::   ../../../assets/plot_scatter.png
            
            """
            
            assert len(x_axis_data) == len(y_axis_data), \
                f"Length of data and categories should be equal; found {len(x_axis_data)} and {len(y_axis_data)}"
            assert len(x_axis_data) == len(marker_shapes), \
                (
                    f"Length of marker shapes and categories should be equal;found {len(x_axis_data)} and {len(marker_shapes)}")

            colors = optional_params.get('colors', gen_hex_from_cm('Pastel1'))

            # Initialize an empty list for the Plotly traces
            traces = []

            # Generate the scatter plot for each series
            for i, (x, y, shape) in enumerate(zip(x_axis_data, y_axis_data, marker_shapes)):
                traces.append(go.Scatter(
                    x=x,
                    y=y,
                    mode='markers',
                    marker=dict(
                        symbol=shape,
                        size=10,  # Or any other size logic
                        color=colors[i % len(colors)],
                        line=dict(width=2, color='DarkSlateGrey')
                    ),
                    name=f"Group {i + 1}"
                ))

            # Create the layout
            layout = go.Layout(
                title=optional_params.get('title', ''),
                xaxis=dict(title=optional_params.get('xlabel', '')),
                yaxis=dict(title=optional_params.get('ylabel', '')),
                template=optional_params.get('theme', 'plotly'),
                legend=dict(yanchor="top", y=0.99, xanchor="left", x=0.01)
            )

            # Create the figure with all traces and the specified layout
            fig = go.Figure(data=traces, layout=layout)

            return fig

        def line_plot(input_vals: List, predicted_line: List, actual_line: List, **optional_params) -> go.Figure:
            """
                Creates a line plot.

                :param input_vals: The values of the inputs
                :param predicted_line: The predicted line
                :param actual_line: The actual line
                :param optional_params:
                    * xlabel: Label for the x-axis. Defaults to an empty string if not specified.
                    * ylabel: Label for the y-axis. Defaults to an empty string if not specified.
                    * title: Title of the plot. Defaults to empty strin if not specified.
                    * theme: Theme for the plot. Defaults to 'plotly' if not specified.
                :return: A plotly go Figure

                :example:
                
                    >>> dash_manager = DashManager()
                    >>> x = ['8:00 PM', '9:00 PM', '10:00 PM', '12:00 PM', '1:00 AM', '3:00 AM', '4:00 AM', '5:00 AM']
                    >>> yp = [30, 30, 33, 28, 34, 29, 33, 32]
                    >>> y = [30, 30, 29, 33, 32, 29, 31, 37]
                    >>> timeseries = PlotGenerator.Timeseries()
                    >>> fig = timeseries.line_plot(x, yp, y, title='Line Chart', xlabel='Time', \
                    ylabel='Value')
                    >>> dash_manager.add_figure(fig)
                    >>> dash_manager.run()

                    .. image::   ../../../assets/plot_baseline.png
                
            """
            
            if len(input_vals) != len(predicted_line) or len(input_vals) != len(actual_line):
                raise ValueError("xAxisvalue, predictedLine, and actualLine must have the same length.")

            # Create a new figure
            fig = go.Figure()

            # Add actual line
            fig.add_trace(go.Scatter(
                x=input_vals,
                y=actual_line,
                mode='lines',
                name='Actual',
                line=dict(color=optional_params.get('actualLineColor', DefaultColors.primary_line),
                          dash='dash')
            ))

            # Add predicted line
            fig.add_trace(go.Scatter(
                x=input_vals,
                y=predicted_line,
                mode='lines',
                name='Predicted',
                line=dict(color=optional_params.get('predictedLineColor', DefaultColors.secondary_line))
            ))

            # Update layout
            fig.update_layout(
                title=optional_params.get('title', ''),

        @staticmethod
        def stack_bar(categories: List, series_data: List[List], series_names: List,
                      **optional_params) -> go.Figure:
            """
                Creates a stacked bar plot.

                :param categories: The categories for the x-axis
                :param series_data: A list of lists containing data for each series
                :param series_names: A list of names for each series
                :param optional_params:
                    * colors: A list of colors for each category. Defaults to Pastel1 colormap if not specified.
                    * xlabel: Label for the x-axis. Defaults to an empty string if not specified.
                    * ylabel: Label for the y-axis. Defaults to an empty string if not specified.
                    * title: Title of the plot. Defaults to empty strin if not specified.
                    * theme: Theme for the plot. Defaults to 'plotly' if not specified.
                :return: A plotly figure object

                :example:
                
                    >>> dash_manager = DashManager()
                    >>> timeseries = PlotGenerator.Timeseries()
                    >>> categories = ['Category A', 'Category B', 'Category C', 'Category D']
                    >>> series_data = [\
                        [20, 15, 25, 10],\
                        [30, 25, 15, 20],\
                        [10, 20, 30, 25]\
                    ]
                    >>> series_names = ['Series 1', 'Series 2', 'Series 3']
                    >>> # Optional parameters
                    >>> optional_params = {\
                        'title': 'Stacked Bar Chart Example',\
                        'xlabel': 'Categories',\
                        'ylabel': 'Value',\
                        'theme': 'plotly',\
                    }
                    >>> colors = ['red', 'green', 'blue']
                    >>> # Create the stacked bar chart
                    >>> fig = timeseries.stack_bar(categories, series_data, series_names, colors, **optional_params)
                    >>> dash_manager.add_figure(fig)
                    >>> dash_manager.run()

                .. image::   ../../../assets/plot_stack_bar.png
                
            """
            
            assert len(series_data) == len(series_names), \
                f"Length of data and categories should be equal; found {len(series_data)} and {len(series_names)}"
            if not all(len(l) == len(categories) for l in series_data):
                raise ValueError(f'Not all lists have same length of categories {len(categories)}')

            colors = optional_params.get('colors', gen_hex_from_cm('Pastel1'))

            # Create a new figure
            fig = go.Figure()

            for i, data in enumerate(series_data):
                # Add bars for each series
                fig.add_trace(go.Bar(
                    x=categories,
                    y=data,
                    name=series_names[i],
                    marker_color=colors[i % len(colors)]
                ))

            # Update layout to stack the bars
            fig.update_layout(
                barmode='stack',
                title=optional_params.get('title', ''),
                xaxis=dict(title=optional_params.get('xlabel', '')),
                yaxis=dict(title=optional_params.get('ylabel', '')),
                template=optional_params.get('theme', 'plotly'),
                legend=dict(yanchor="top", y=0.99, xanchor="left", x=0.01)
            )

            return fig

class DashManager:
    """
        A utility class for managing and displaying Plotly Dash figures in a web application.

        This class simplifies the process of creating and displaying interactive Plotly Dash figures
        in a Dash web application. It provides methods for adding figures, setting up the layout,
        and running the Dash server.


        .. code-block:: python

            dash_manager = DashManager()
            fig1 = create_plotly_figure_1()
            fig2 = create_plotly_figure_2()
            dash_manager.add_figure(fig1)
            dash_manager.add_figure(fig2)
            dash_manager.run()

        :ivar app: The Dash application instance.
        :vartype app: dash.Dash
        :ivar figures: A list of Plotly figures to be displayed in the application.
        :vartype figures: list
        
    """

    def __init__(self):
        """
            Initialize a DashManager instance.

            This constructor initializes the DashManager by creating a Dash application instance
            and an empty list to store Plotly figures.
            
        """
        
        self.app = Dash(__name__)
        self.figures = []

    def add_figure(self, figure):
        """
            Add a Plotly figure to the list of figures.

            :param figure: A Plotly figure to be added to the list of figures.
            :type figure: plotly.graph_objs.Figure
            
        """
        
        self.figures.append(figure)

    def setup_layout(self):
        """
            Set up the layout of the Dash application.

            This method configures the layout of the Dash application by creating a Div element
            containing dcc.Graph components for each figure added to the list of figures.
            
        """
        
        self.app.layout = html.Div([
            dcc.Graph(figure=fig) for fig in self.figures
        ])

    def run(self):
        """
            Start the Dash server to run the application.

            This method starts the Dash server to run the application with the configured layout.
            It also enables debugging mode to aid in development.
            
        """
        
        self.setup_layout()
        self.app.run_server(debug=True)
