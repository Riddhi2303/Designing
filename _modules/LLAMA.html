    @performance.timeit
    def positional_encoding(self, position: int, d_model: int):
        """
        This method generates a matrix of angles for positional encoding.

        :param position: The position of the node in the tree.
        :param d_model: The dimension of the model.

        :return: The positional encoding.

        :example:

            >>> from mycelium.model import Transformer
            >>> Transformer.positional_encoding(1, 2)

        """

        # Generate a matrix of angles for positional encoding
        angle_rads = self.get_angles(tf.range(position)[:, tf.newaxis],
                                tf.range(d_model)[tf.newaxis, :],
                                d_model)
        
        # Apply sin function to even indices in the array (2i)
        angle_rads[:, 0::2] = tf.math.sin(angle_rads[:, 0::2])
        
        # Apply cos function to odd indices in the array (2i+1)
        angle_rads[:, 1::2] = tf.math.cos(angle_rads[:, 1::2])
        
        # Add an extra dimension to match the batch size
        pos_encoding = angle_rads[tf.newaxis, ...]
        
        # Return the positional encoding
        return pos_encoding

    @performance.timeit
    def get_angles(self, pos: int, i: int, d_model: int):
        """This method computes the angle rates for positional encoding.

            :param pos: The position of the node in the tree.
            :param i: The dimension of the model.
            :param d_model: The dimension of the model.

            :return: The angle rates for positional encoding.

            :example:

                >>> from mycelium.model import Transformer
                >>> Transformer.get_angles(1, 2, 3)

        """

        # Calculate the angle rates for positional encoding
        angle_rates = 1 / tf.pow(10000, (2 * (i//2)) / tf.cast(d_model, tf.float32))
        return pos * angle_rates

    @performance.timeit
    def forward(self, X: tf.Tensor, W_q: tf.Tensor, W_k: tf.Tensor, W_v: tf.Tensor):
        """This method computes the output of the self-attention mechanism in the Transformer.

            :param X: The input.
            :param W_q: The query matrix.
            :param W_k: The key matrix.
            :param W_v: The value matrix.

            :return: The output of the self-attention mechanism.

            :example:
            
                >>> from mycelium.model import Transformer
                >>> Transformer.forward(X, W_q, W_k, W_v)

        """

        # Compute Q, K, V
        Q = tf.matmul(X, W_q)
        K = tf.matmul(X, W_k)
        V = tf.matmul(X, W_v)

        # Compute attention scores
        scores = tf.matmul(Q, K, transpose_b=True) / tf.sqrt(float(K.shape[-1]))

        # Apply softmax to get attention weights
        weights = tf.nn.softmax(scores)

        # Compute output
        out = tf.matmul(weights, V)
        
        return out, weights

class RMSNorm(tf.keras.layers.Layer):
    def __init__(self, dim: int, eps: float = 1e-6, **kwargs):
        """
        Initialize the RMSNorm normalization layer.
        
        :param dim: The dimension of the input tensor.
        :param eps: A small value added to the denominator for numerical stability. Default is 1e-6.
        
        """
        
        super().__init__(**kwargs)
        self.eps = eps
        self.dim = dim

    def build(self, input_shape):
        """Create the weight variable during the build of the layer."""
        self.weight = self.add_weight(
            shape=(self.inpu,),
            initializer="ones",  # initialize as ones
            trainable=True,  # trainable parameter
        )

    def call(self, x):
        """
        Forward pass through the RMSNorm layer.
        
        :param x: The input tensor.
        :return: The output tensor after applying RMSNorm.

        """
        # Compute the norm of the input tensor
        norm = x * tf.math.rsqrt(tf.reduce_mean(tf.square(x), axis=-1, keepdims=True) + self.eps)
        
        # Return the normalized tensor, scaled by the learned weight
        return norm * self.weight

def precompute_freqs_cis(dim: int, end: int, theta: float = 10000.0):
    """
    Precompute the frequency tensor for complex exponentials (cis) with given dimensions.

    This function calculates a frequency tensor with complex exponentials using the given dimension 'dim'
    and the end index 'end'. The 'theta' parameter scales the frequencies.
    The returned tensor contains complex values in complex64 data type.
    
    :param dim: Dimension of the frequency tensor.
    :param end: End index for precomputing frequencies.
    :param theta: Scaling factor for frequency computation. Default is 10000.0.

    :return: The frequency tensor.
    
    """
    
    freqs = 1.0 / (theta ** (tf.range(0, dim, 2)[: (dim // 2)] / tf.cast(dim, tf.float32)))
    t = tf.range(end)
    freqs = tf.tensordot(t, freqs, axes=0)
    freqs_cis = tf.complex(tf.ones_like(freqs), freqs)
    return freqs_cis

@performance.timeit
def reshape_for_broadcast(freqs_cis: tf.Tensor, x: tf.Tensor):
    """
    Reshape the frequency tensor for broadcasting compatibility with the input tensor.

    This function reshapes the frequency tensor 'freqs_cis' to be compatible for broadcasting with the input tensor 'x'.
    The reshaping is done based on the dimensions of the input tensor.

    :param freqs_cis: Frequency tensor to be reshaped.
    :param x: Input tensor to determine the reshaping.
    
    :return: Reshaped frequency tensor.
    
    """

    ndim = len(x.shape)
    assert 0 <= 1 < ndim
    assert freqs_cis.shape == (x.shape[1], x.shape[-1])
    return tf.reshape(freqs_cis, [d if i == 1 or i == ndim - 1 else 1 for i, d in enumerate(x.shape)])

@performance.timeit
def apply_rotary_emb(xq: tf.Tensor, xk: tf.Tensor, freqs_cis: tf.Tensor):
    """
    Apply rotary embeddings to the input tensors.

    This function applies rotary embeddings to the input tensors 'xq' and 'xk' using the frequency tensor 'freqs_cis'.
    The input tensors are first reshaped and converted to complex numbers, then the frequency tensor is reshaped for broadcasting.
    The rotary embeddings are applied by multiplying the input tensors with the frequency tensor.
    
    :param xq: Input tensor for query.
    :param xk: Input tensor for key.
    :param freqs_cis: Frequency tensor for rotary embeddings.
    
    :return: Tuple containing the query and key tensors with applied rotary embeddings.
    
    :example:
    
        >>> xq = tf.random.normal((1, 2, 4))
        >>> xk = tf.random.normal((1, 2, 4))
        >>> freqs_cis = precompute_freqs_cis(4, 2)
        >>> apply_rotary_emb(xq, xk, freqs_cis)
        (<tf.Tensor: shape=(1, 2, 4), dtype=float32, numpy= ... >, <tf.Tensor: shape=(1, 2, 4), dtype=float32, numpy= ... >)

    """
    
    xq_ = tf.complex(xq[..., 0::2], xq[..., 1::2])
    xk_ = tf.complex(xk[..., 0::2], xk[..., 1::2])
    freqs_cis = reshape_for_broadcast(freqs_cis, xq_)
    return (tf.cast(tf.math.real(xq_ * freqs_cis), xq.dtype), tf.cast(tf.math.real(xk_ * freqs_cis), xk.dtype))

@performance.timeit
def repeat_kv(x: tf.Tensor, n_rep: int) -> tf.Tensor:
    """
    Repeat the key and value tensors for multiple heads.
    
    This function repeats the key and value tensors 'n_rep' times for multiple heads.
    The input tensor 'x' is expected to have the shape (batch_size, seq_len, n_kv_heads, head_dim).
    
    :param x: Input tensor for key or value.
    :param n_rep: Number of repetitions for the key and value tensors.
    
    :return: The repeated key or value tensor.
    
    :example:
    
        >>> x = tf.random.normal((1, 2, 4, 4))
        >>> repeat_kv(x, 2)
        <tf.Tensor: shape=(1, 2, 8, 4), dtype=float32, numpy= ... >
        
    """
    
    if n_rep == 1: return x
    bs, slen, n_kv_heads, head_dim = x.shape
    x = tf.expand_dims(x, axis=3)
    x = tf.tile(x, [1, 1, 1, n_rep, 1])
    return tf.reshape(x, [bs, slen, n_kv_heads * n_rep, head_dim])
